[GXY]Baby nc(判断数大小) 

from pwn import *
context.log_level='debug'
r=remote("183.129.189.60",10014)
flag='34028236692093846346337460'
for a in range(10):
    for i in range(10):
    	guess=str(flag)+str(i)+(27-1-len(flag))*'0'
        r.sendline(guess)
        result=r.recv(10)
        if 'big' in result:
            flag+=str(i-1)
            break
        if i==9 and 'small' in result:
            flag+='9'
            break
    print(flag)
