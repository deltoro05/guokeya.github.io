使用private关键字。使age属性只能由类内调用。

如果没有编写任意的构造方法。那么编译器会给你一个默认的构造方法。啥也不敢

import java.util.*;
public class index{
	public static void main(String[] args){
		Scanner sc =new Scanner(System.in);
		int FirstNum = sc.nextInt();
		int SecondNum = sc.nextInt();
		String op = sc.next();
		switch(op){
			case "+":
				System.out.println(FirstNum+"+"+SecondNum+"="+(FirstNum+SecondNum));
				break;
			case "-":
				System.out.println(FirstNum+"-"+SecondNum+"="+(FirstNum-SecondNum));
				break;
			case "*":
				System.out.println(FirstNum+"*"+SecondNum+"="+(FirstNum*SecondNum));
				break;
			case "/":
				System.out.println(FirstNum+"/"+SecondNum+"="+(FirstNum/SecondNum));
				break;
			default :
				System.out.println("Invaild op");
		}

	}
}

import java.io.*;
public class Person implements Serializable{

	}

	public Person(){

seri.java

import java.io.*;
public class seri{
	public static void main(String[] args) throws Exception{
        #需要有Exception来接受错误
		ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream("Person.txt"));
        #对象输出流。ObjectOutput/FileOutPut/Person.txt
		oos.writeObject(new Person("name",18));
        #写入文件
		oos.close();
        #释放文件
		ObjectInputStream oss = new ObjectInputStream(new FileInputStream("Person.txt"));
        #对象输入流。ObjectInput/FileInput/Person.txt
		Person o = (Person)oss.readObject();
        #读取对象。类型/名称/类
		oss.close();
		System.out.println(o);
        //输出类。触发toString。感觉和PHP那个魔法函数一样
	}
}

static关键字：静态关键字
静态优先于非静态加载到内存中(静态优先于对象进入到内存中)
被static修饰的成员变量不能被序列化。序列化的都是对象
transient关键字:瞬态关键字
加了transient关键字。就不会被反序列化。
