static关键字：静态关键字
静态优先于非静态加载到内存中(静态优先于对象进入到内存中)
被static修饰的成员变量不能被序列化。序列化的都是对象

public class main{
	static String name;
	// 静态代码块
	static {
		System.out.println(name + "静态代码块");
	}
	// 非静态代码块
	{
		System.out.println(name + "非静态代码块");
	}
	public main(String a) {
		name = a;
		System.out.println(name + "构造方法");
	}
	public void method() {
		System.out.println(name + "成员方法");
	}
	public static void main(String[] args) {
		main s1;// 声明的时候就已经运行静态代码块了
		main s2 = new main("s2");// new的时候才会运行构造方法
		main s3 = new main("s3");
		s3.method();// 只有调用的时候才会运行
	}
}

这里就一个口诀。
对于成员变量：编译看左边。运行看左边
对于成员方法：编译看左边。运行看右边
如果你左边编译不成功。就无法执行
在main.java中。new了一个zi类。
调用z的num变量。那么输出的却是fu类的变量。

调用z的method方法。输出的是zi类的method方法。因为编译看左边。fu类中存在method方法。成功编译。然后运行看右边。调用子类的method方法

调用z的methodfu。成功输出fu类的methodfu方法。因为编译。fu类中存在methodfu方法。调用zi类的methodfu方法。由于不存在。就会往父类去找。找到methodfu方法。从而调用

        Statement stmt = null;
        Connection conn = null;

            //获取数据库连接对象。jdbc:数据库类型://IP:端口/数据库名?时间          用户名     密码
            //String sql = "update user set password='tetetes' where username='admin'";
            String sql = "delete from user where username='guoke'";
            //定义sql语句
            stmt= conn.createStatement();
            //stmt是一个用来执行sql的对象
            int count = stmt.executeUpdate(sql);
            //调用执行sql的对象中的executeUpdate这个方法。sql作为参数传入。返回值是个int。印象行数
            if (count>0){
                System.out.println("影响行数："+count);
                //通过判断返回值。判断程序是否执行成功
            }else{
                System.out.println("未成功更新。请检查操作");
            }
        } catch (SQLException e){
            e.printStackTrace();
            //如果报错就输出
        }finally {
            //没报错。就开始释放连接。注意：这里报错了程序还会正常执行。但是连接对象缺没正常赋值。所以得先判断下是否赋值。再释放
            if (stmt !=null){
                //万一有啥特殊原因。释放失败就打印错误信息
                try{
                    stmt.close();
                }catch (SQLException e){
                    e.printStackTrace();
                }
            }
            if (conn !=null){
                try {
                    conn.close();
                }catch (SQLException e){
                    e.printStackTrace();
                }
            }
        }
    }
}

package cn.itcast.jdbc;

import java.io.FileReader;
import java.io.IOException;
import java.net.URL;
import java.sql.*;
import java.util.Properties;

public class jdbcutils {
    private static String url;
    private static String username;
    private static String password;
    private static String driver;
    private static Connection conn;
    private static PreparedStatement stmt;
    private static String sql;
    private static ResultSet rs;
    static {
        try{
            ClassLoader classloader = jdbcutils.class.getClassLoader();
            URL res = classloader.getResource("jdbc.properties");
            String Path = res.getPath();
            Properties pro = new Properties();
            pro.load(new FileReader(Path));
            url = pro.getProperty("url");
            username = pro.getProperty("username");
            password = pro.getProperty("password");
            driver = pro.getProperty("driver");
            Class.forName(driver);
        }catch(IOException e){
            e.printStackTrace();
        }catch(ClassNotFoundException e) {
            e.printStackTrace();
        }
    }
    private static Connection Getconn() throws Exception{
        return DriverManager.getConnection(url,username,password);
    }

    public static void dbclose()throws Exception{
        if (conn != null) {
            try {
                conn.close();
            }catch(Exception e){
                e.printStackTrace();
            }
        }
        if (stmt != null) {
            try {
                stmt.close();
            }catch(Exception e){
                e.printStackTrace();
            }
        }
        System.out.println("成功关闭数据库连接");
    }
    public static boolean UserLogin(String username, String password) throws Exception{
        sql = "select * from user where username=? and password=?";
        conn = Getconn();
        stmt = conn.prepareStatement(sql);
        stmt.setString(1,username);
        stmt.setString(2,password);
        rs = stmt.executeQuery();
        return rs.next();
    }
}
