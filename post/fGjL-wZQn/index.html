$parts = explode(".", $name);

$ext = array_pop($parts);

if (empty($parts[0])) {
    array_shift($parts);
    #如果数组第一个值为空。就删掉。
}
$name = implode(".", $parts);
#组合。
move_uploaded_file($tmp_name, $userdir . $name . "." . $ext);
