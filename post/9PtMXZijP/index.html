HTTP模块

var http = require('http');
#导入http模块
http.createServer(function(request,response){
    #创建http对象。包含两个参数。request和response
	if(request.url='/'){
    #如果请求的是/。就进行以下操作
	response.writeHead(200,{'COntent-Type':'text/html;charset=utf-8'})
    #返回头
	response.write('hello world');
    #返回文本
	console.log(request.url);
    #在终端输出
	response.end();
    #返回头结束
	}
}).listen(808);
#监听本地808端口

调用函数。

var http = require('http');
http.createServer(function(request,response){
	if(request.url='/'){
	response.writeHead(200,{'COntent-Type':'text/html;charset=utf-8'})
	test(response,'test')
    #调用自定义的test函数
	response.end();
	}
}).listen(808);
function test(response,test){
    #传入response和test
	response.write(test)
}

调用其他文件中的函数

index.js

var http = require('http');
var func = require('./func.js')
http.createServer(function(request,response){
	if(request.url='/'){
	response.writeHead(200,{'COntent-Type':'text/html;charset=utf-8'})
	func.test(response,'test')
    #当func.js中定义了exports。可以写成func(response,'test')
    #调用自定义的test函数
	response.end();
	}
}).listen(808);

func.js

function test(response,test){
    #传入response和test
	response.write(test)
}
#当导入此模块时。自动执行test函数
#module.exports = test;

当有多个自定义函数时。也可以不写exports或者多个export

module.exports={
    test1:function(response,name){
        response.write(name);
    },
    test2:function(response,age){
        response.write(age);
    }
}
